Quick Usage Example

// Import MongoSQL
const MongoSQL = require('mongosql');
const db = new MongoSQL('mongodb://localhost:27017/mydb');

// SELECT like SQL
const users = await db.query('SELECT * FROM users WHERE age > 25');

// INSERT
await db.query("INSERT INTO users (name, age) VALUES ('John Doe', 30)");

// UPDATE
await db.query("UPDATE users SET age = 31 WHERE name = 'John Doe'");

// DELETE
await db.query("DELETE FROM users WHERE age < 18");
